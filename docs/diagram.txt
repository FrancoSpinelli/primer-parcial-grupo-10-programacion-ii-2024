}

  +Vendedor(Persona)
  
  +listarReservasPendientes():void

}

}

  +calcularPrecioTotal(int): float
  +verListadoAutos(): void
  +entregarAuto(): void

}

class Gasolina {
  -capacidadMaxima: float
  -cantidad: float
